import pandas as pd

# YoY growth helper

def yoy_growth(curr, prev):
    return (curr / prev) - 1.0

# Example usage
revenue = pd.Series({"Q2_FY24": 1042.0, "Q2_FY25": 1171.0})
yoy = yoy_growth(revenue["Q2_FY25"], revenue["Q2_FY24"]) * 100

print(f"YoY Revenue Growth: {yoy:.2f}%")
